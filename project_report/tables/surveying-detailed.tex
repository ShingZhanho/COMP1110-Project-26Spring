% Google Maps
\newcommand{\gMapsETA}{Provided ETA with multiple methods, calculated dynamically with the current situation. }
\newcommand{\gMapsFairDisplay}{No fare display.}
\newcommand{\gMapsTransferHandling}{Able to handle any transfer, even if some distance needs to be walked.}
\newcommand{\gMapsAccessibilityInfo}{Provided accessibility mode to avoid stairs and prioritize the lift.}
\newcommand{\gMapsOfflineAvailability}{No route planning function when offline.}

% Citymapper

\newcommand{\citymapperFairDisplay}{{Provided adult fares for multiple methods.

Taxi fare is estimated by driving time.}}

\newcommand{\citymapperAccessibilityInfo}{No accessibility info.}
\newcommand{\citymapperOfflineAvailability}{Unable to use it offline.}

% MTR Mobile
\newcommand{\mtrMobileETA}{Provided ETA with multiple methods, estimated by maximum waiting time.}
\newcommand{\mtrMobileFairDisplay}{Adult fare is displayed, while others can be checked.}
\newcommand{\mtrMobileTransferHandling}{Able to handle any transfer, only within MTR stations or light rail.}
\newcommand{\mtrMobileAccessibilityInfo}{Able to display accessibility facilities for multiple kinds of accessibility needs, such as those of the visually impaired.}
\newcommand{\mtrMobileOfflineAvailability}{No route planning function when offline.}

% HKUBUS.app
\newcommand{\hkubusETA}{ETA must be checked by checking different paths one by one.}
\newcommand{\hkubusFairDisplay}{Adult fare is displayed separately for different paths, and you need to sum them up to get a total.}
\newcommand{\hkubusTransferHandling}{At most one transfer is allowed. Able to handle at most 500m walking distance.}
\newcommand{\hkubusAccessibilityInfo}{No accessibility info.}
\newcommand{\hkubusOfflineAvailability}{Unable to use it offline.}
